namespace = confrontations_{{faction}}

# {{faction}}
country_event = {
    id = confrontations_{{faction}}.1
    hide_window = yes

    is_triggered_only = no

    mean_time_to_happen = { months = 1 }
    trigger = {
        is_variable_set = current_{{faction}}_support
        is_variable_set = current_ideological_purity

        check_variable = {
            which = current_{{faction}}_support
            value > current_ideological_purity
        }
    }

    immediate = {
        export_resource_income_to_variable = { 
            resource = {{faction}}_support 
            variable = current_{{faction}}_support_income
        }

        # One years income
        multiply_variable = {
            which = current_{{faction}}_support_income
            value = 12
        }

        add_resource = { 
            {{faction}}_support = -1
            mult = current_{{faction}}_support_income 
        }

        export_resource_stockpile_to_variable = { 
            resource = {{faction}}_militancy
            variable = current_{{faction}}_militancy
        }

        if = {
            limit = {
                check_variable = {
                    which = current_{{faction}}_militancy
                    value > 75
                }
            }
            country_event = { id = confrontations_{{faction}}.40 }
        }	
        else_if = {
            limit = {
                check_variable = {
                    which = current_{{faction}}_militancy
                    value > 50
                }
            }
            country_event = { id = confrontations_{{faction}}.30 }
        }
        else_if = {
            limit = {
                check_variable = {
                    which = current_{{faction}}_militancy
                    value > 25
                }
            }
            country_event = { id = confrontations_{{faction}}.20 }
        }
        else = {
            country_event = { id = confrontations_{{faction}}.10 }
        }	
    }
}

    }

        name = confrontations_{{faction}}.10.a

        export_resource_income_to_variable = {
            resource = {{resource}}
            variable = current_monthly_{{resource}}_income
        }
        multiply_variable = {
            which = current_monthly_{{resource}}_income
            value = 6
        }
        
        add_resource = {
            {{resource}} = 1
            mult = current_monthly_{{resource}}_income
        }
        add_modifier = {
            modifier = {{faction}}_debates_continue_modifier

        }
    }
    option = {
        name = confrontations_{{faction}}.10.b

        add_modifier = {
            modifier = {{faction}}_debates_stopped_modifier
            years = 3
        }
    }
}
