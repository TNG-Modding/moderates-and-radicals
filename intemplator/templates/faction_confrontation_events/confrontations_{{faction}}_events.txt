country_event = {
    id = confrontations_{{faction}}.1
    hide_window = yes

    is_triggered_only = no

    mean_time_to_happen = { months = 1 }
    trigger = {
        NOT = {
            has_modifier = recent_political_event
        }
        has_modifier = politics
        is_variable_set = current_{{faction}}_support
        is_variable_set = current_ideological_purity

        check_variable = {
            which = current_{{faction}}_support
            value > current_ideological_purity
        }
    }

    immediate = {
        add_modifier = {
            modifier = recent_political_event
            years = 1
        }
        export_resource_income_to_variable = { 
            resource = {{faction}}_support 
            variable = current_{{faction}}_support_income
        }

        # One years income
        multiply_variable = {
            which = current_{{faction}}_support_income
            value = 36
        }

        add_resource = { 
            {{faction}}_support = -1
            mult = current_{{faction}}_support_income 
        }

        export_resource_stockpile_to_variable = { 
            resource = {{faction}}_militancy
            variable = current_{{faction}}_militancy
        }

        if = {
            limit = {
                check_variable = {
                    which = current_{{faction}}_militancy
                    value > 90
                }
            }
            every_pop_faction = {
                limit = {
                    OR = {
                        is_pop_faction_type = gov_{{faction}}
                        is_pop_faction_type = {{faction}}_radicals
                        is_pop_faction_type = {{faction}}
                    }
                }
                random_list = {
                    100 = {
                        pop_faction_event = { id = confrontations_{{faction}}.90 }
                    }
                    
                }
                
            }
        }	
        else_if = {
            limit = {
                check_variable = {
                    which = current_{{faction}}_militancy
                    value > 80

                        is_pop_faction_type = {{faction}}

    id = confrontations_{{faction}}.90

    picture = GFX_evt_open_revolt
    show_sound = event_super_explosion

    immediate = {
        owner = { country_event = { id = confrontations.1 } }
    }

    option = {
        name = confrontations_{{faction}}.90.a
        
        owner = {
            if = {
                limit = { 
                    NOT = { 
                        has_ethic = "ethic_fanatic_{{ethic}}" 
                    } 
                }
                shift_ethic = ethic_{{ethic}}
            }
            
            add_resource = {
                ideological_purity = -25
            }

            add_modifier = {
                modifier = {{faction}}_coup_continue_modifier
                years = 3
            }

            # Kill ruler
            kill_leader = {
                type = ruler
            }

            # Kill rulers of wrong ethic
            if = {
                limit = {
                    count_owned_pop = {
                        count >= 3
                        limit = {
                            NOT = { pop_has_ethic = ethic_{{ethic}} }
                            is_pop_category = ruler
                        }
                    }
                }
                random_owned_pop = {
                    limit = {
                        NOT = { pop_has_ethic = ethic_{{ethic}} }
                        is_pop_category = ruler
                    }
                    kill_pop = yes
                }

                    limit = {
                        NOT = { pop_has_ethic = ethic_{{ethic}} }

                    }
                    kill_pop = yes
                }

                    }
