namespace = confrontations_{{faction}}

situation_event = {
    id = confrontations_{{faction}}.11
    title = confrontations_{{faction}}.11
    desc = confrontations_{{faction}}.11.desc
    hide_window = no

    is_triggered_only = yes

    picture = GFX_evt_assembly_fight
    show_sound = event_factions

    trigger = {
        owner = {
            count_owned_leader = {
                limit = {
                    leader_of_faction = prevprev
                }
                count = 1
            }
        }
    }
    
    immediate = {
        owner = {
            # Assign radical faction
            random_pop_faction = {
                limit = {
                    is_pop_faction_type = radical_{{faction}}
                }
                save_event_target_as = situation_faction
            }
        }
    }

    option = {
        name = confrontations_{{faction}}.11.a

        owner = {
            random_owned_pop = {
                limit = {
                    OR = {
                        is_pop_category = ruler
                        is_pop_category = specialist
                    }
                    NOT = {
                        pop_has_ethic = ethic_{{faction}}
                    }
                }
                pop_change_ethic = ethic_{{faction}}
            }
            random_owned_pop = {
                limit = {
                    OR = {
                        is_pop_category = ruler
                        is_pop_category = specialist
                    }
                    NOT = {
                        pop_has_ethic = ethic_{{faction}}
                    }
                }
                pop_change_ethic = ethic_{{faction}}
            }
        }
    }
    option = {
        name = confrontations_{{faction}}.11.b

        owner = {
            add_modifier = {
                modifier = {{faction}}_speech_stopped_modifier
                years = 3
            }
        }
    }
}
