country_event = {
    id = confrontations_supremacist.1
    hide_window = yes

    is_triggered_only = no

    mean_time_to_happen = { months = 1 }
    trigger = {
        NOT = {
            has_modifier = recent_political_event
        }
        has_modifier = politics
        is_variable_set = current_supremacist_support
        is_variable_set = current_ideological_purity

        check_variable = {
            which = current_supremacist_support
            value > current_ideological_purity
        }
    }

    immediate = {
        add_modifier = {
            modifier = recent_political_event
            years = 1
        }
        export_resource_income_to_variable = { 
            resource = supremacist_support 
            variable = current_supremacist_support_income
        }

        # One years income
        multiply_variable = {
            which = current_supremacist_support_income
            value = 36
        }

        add_resource = { 
            supremacist_support = -1
            mult = current_supremacist_support_income 
        }

        export_resource_stockpile_to_variable = { 
            resource = supremacist_militancy
            variable = current_supremacist_militancy
        }

        if = {
            limit = {
                check_variable = {
                    which = current_supremacist_militancy
                    value > 90
                }
            }
            every_pop_faction = {
                limit = {
                    OR = {
                        is_pop_faction_type = gov_supremacist
                        is_pop_faction_type = supremacist_radicals
                        is_pop_faction_type = supremacist
                    }
                }
                random_list = {
                    100 = {
                        pop_faction_event = { id = confrontations_supremacist.90 }
                    }
                    
                }
                
            }
        }	
        else_if = {
            limit = {
                check_variable = {
                    which = current_supremacist_militancy

    }

    option = {
        name = confrontations_supremacist.90.a
        
        owner = {
            if = {
                limit = { 
                    NOT = { 
                        has_ethic = "ethic_fanatic_xenophobe" 
                    } 
                }
                shift_ethic = ethic_xenophobe
            }
            
            add_resource = {
                ideological_purity_current_value = -25

                modifier = supremacist_coup_continue_modifier

            kill_leader = {
                type = ruler
            }

            if = {
                limit = {
                    count_owned_pop = {

                            NOT = { pop_has_ethic = ethic_xenophobe }

                        }
                    }
                }
                random_owned_pop = {
                    limit = {
                        NOT = { pop_has_ethic = ethic_xenophobe }
                        is_pop_category = ruler
                    }
                    kill_pop = yes
                }

                    limit = {
                        NOT = { pop_has_ethic = ethic_xenophobe }

                    }
                    kill_pop = yes
                }

                        is_pop_category	= ruler
                    }

                }
