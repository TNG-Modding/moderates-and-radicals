namespace = traditionalist_confrontations

# traditionalist
country_event = {
    id = traditionalist_confrontations.1
    hide_window = yes

    is_triggered_only = no

    mean_time_to_happen = { months = 1 }
    trigger = {
        is_variable_set = current_traditionalist_support
        is_variable_set = current_ideological_purity

        check_variable = {
            which = current_traditionalist_support
            value > current_ideological_purity
        }
    }

    immediate = {
        export_resource_income_to_variable = { 
            resource = traditionalist_support 
            variable = current_traditionalist_support_income
        }

        # One years income
        multiply_variable = {
            which = current_traditionalist_support_income
            value = 12
        }

        add_resource = { 
            traditionalist_support = -1
            mult = current_traditionalist_support_income 
        }

        export_resource_stockpile_to_variable = { 
            resource = traditionalist_militancy
            variable = current_traditionalist_militancy
        }

        if = {
            limit = {
                check_variable = {
                    which = current_traditionalist_militancy
                    value > 75
                }
            }
            country_event = { id = traditionalist_confrontations.40 }
        }	
        else_if = {
            limit = {
                check_variable = {
                    which = current_traditionalist_militancy
                    value > 50
                }
            }
            country_event = { id = traditionalist_confrontations.30 }
        }
        else_if = {
            limit = {
                check_variable = {
                    which = current_traditionalist_militancy
                    value > 25
                }
            }
            country_event = { id = traditionalist_confrontations.20 }
        }
        else = {
            country_event = { id = traditionalist_confrontations.10 }
        }	
    }
}
