namespace = traditionalist_militancy_desired_resource_events

country_event = {
	id = traditionalist_militancy_desired_resource_events.1
	hide_window = yes
	is_triggered_only = no

	mean_time_to_happen = {
		days = 30
	}

	trigger = {
		has_modifier = politics
	}

	immediate = {
		# Get the current production
		export_resource_income_to_variable = {
			resource = traditionalist_militancy_desired_value_change
			variable = current_traditionalist_militancy_desired_value_change
		}

		# Define new desired point
		set_variable = {
			which = traditionalist_militancy_new_desired_value
			value = 5
		}
		change_variable = {
			which = traditionalist_militancy_new_desired_value
			value = current_traditionalist_militancy_desired_value_change
		}

		# Clamp desired value to 0 inclusive to 99 inclusive
		if = {
			limit = {
				check_variable = {
					which = traditionalist_militancy_new_desired_value
					value > 99
				}
			}
			set_variable = {
				which = traditionalist_militancy_new_desired_value
				value = 99
			}
		}
		else_if = {
			limit = {
				check_variable = {
					which = traditionalist_militancy_new_desired_value
					value < 0
				}
			}
			set_variable = {
				which = traditionalist_militancy_new_desired_value
				value = 0
			}
		}
		# traditionalist_militancy_new_desired_value is now the new desired point
		
		# Old desired value
		export_resource_stockpile_to_variable = {
			resource = traditionalist_militancy_desired_value
			variable = current_traditionalist_militancy_desired_value
		}

		# Determine the delta needed to achieve new desired value
		set_variable = {
			which = traditionalist_militancy_desired_value_delta
			value = traditionalist_militancy_new_desired_value 
		}
		subtract_variable = {
			which = traditionalist_militancy_desired_value_delta
			value = current_traditionalist_militancy_desired_value
		}
		# traditionalist_militancy_desired_value_delta is now the delta between new and old desired point

		# Set desired value to new desired value
		add_resource = {
			traditionalist_militancy_desired_value = 1
			mult = traditionalist_militancy_desired_value_delta
		}

		# Get current value of traditionalist_militancy
		export_resource_stockpile_to_variable = {
			resource = traditionalist_militancy_current_value
			variable = current_traditionalist_militancy_current_value
		}

		set_variable = {
			which = delta_between_current_desired_traditionalist_militancy
			value = traditionalist_militancy_new_desired_value
		}
		subtract_variable = {
			which = delta_between_current_desired_traditionalist_militancy
			value = current_traditionalist_militancy_current_value
		}
		# delta_between_current_desired_traditionalist_militancy now represents the difference between the values

		# If we are close, remove modifiers, if we are far, add trending up or trending down
		if = {
			limit = {
				check_variable = {
					which = delta_between_current_desired_traditionalist_militancy
					value <= 1
				}
				check_variable = {
					which = delta_between_current_desired_traditionalist_militancy
					value >= -1
				}
			}
			remove_modifier = traditionalist_militancy_current_value_trending_up
			remove_modifier = traditionalist_militancy_current_value_trending_down
		}
		else = {
			if = {
				limit = {
					check_variable = {
						which = traditionalist_militancy_new_desired_value
						value >= current_traditionalist_militancy_current_value
					}
				}
				add_modifier = {
					modifier = traditionalist_militancy_current_value_trending_up
					days = -1
				}
			}
			else = {
				add_modifier = {
					modifier = traditionalist_militancy_current_value_trending_down
					days = -1
				}
			}
		}

		# create_message = {
		# 	type = traditionalist_militancy_values_updated
		# 	localization = traditionalist_militancy_values_updated_message
		# 	days = 5
		# 	target = root
		# }
	}	
}
