namespace = confrontations_prosperity

# prosperity
country_event = {
    id = confrontations_prosperity.1
    hide_window = yes

    is_triggered_only = no

    mean_time_to_happen = { months = 1 }
    trigger = {
        is_variable_set = current_prosperity_support
        is_variable_set = current_ideological_purity

        check_variable = {
            which = current_prosperity_support
            value > current_ideological_purity
        }
    }

    immediate = {
        export_resource_income_to_variable = { 
            resource = prosperity_support 
            variable = current_prosperity_support_income
        }

        # One years income
        multiply_variable = {
            which = current_prosperity_support_income
            value = 12
        }

        add_resource = { 
            prosperity_support = -1
            mult = current_prosperity_support_income 
        }

        export_resource_stockpile_to_variable = { 
            resource = prosperity_militancy
            variable = current_prosperity_militancy
        }

        if = {
            limit = {
                check_variable = {
                    which = current_prosperity_militancy
                    value > 75
                }
            }
            country_event = { id = confrontations_prosperity.40 }
        }	
        else_if = {
            limit = {
                check_variable = {
                    which = current_prosperity_militancy
                    value > 50
                }
            }
            country_event = { id = confrontations_prosperity.30 }
        }
        else_if = {
            limit = {
                check_variable = {
                    which = current_prosperity_militancy
                    value > 25
                }
            }
            country_event = { id = confrontations_prosperity.20 }
        }
        else = {
            country_event = { id = confrontations_prosperity.10 }
        }	
    }
}

# Political debate
country_event = {
    id = confrontations_prosperity.10
    title = confrontations_prosperity.10
    desc = confrontations_prosperity.10.desc
    hide_window = no

    is_triggered_only = yes

    picture = GFX_evt_assembly_fight
    show_sound = event_super_explosion

    hidden_effect = {
        country_event = { id = confrontations.1 }
    }

    option = {
        name = confrontations_prosperity.10.a

        hidden_effect = {
            export_resource_income_to_variable = {
                resource = consumer_goods
                variable = current_monthly_consumer_goods_income
            }
            multiply_variable = {
                which = current_monthly_consumer_goods_income
                value = 6
            }
        }
        add_resource = {
            consumer_goods = 1
            mult = current_monthly_consumer_goods_income
        }
        add_modifier = {
            modifier = prosperity_debates_continue_modifier
            years = 3
        }
    }
    option = {
        name = confrontations_prosperity.10.b

        add_modifier = {
            modifier = prosperity_debates_stopped_modifier
            years = 3
        }
    }
}
