# Diversity of Discussion Events
###############################

namespace = diversity_of_discussion

# High Diversity of Discussion
country_event = {
	id = diversity_of_discussion.1
	title = diversity_of_discussion.1.title

	hide_window = yes
	is_triggered_only = no
	
	trigger = {
		is_country_type = default
		
		owner = { 
			NOT = {
				has_modifier = high_diversity_of_discussion
			}
			owner = { has_country_flag = first_faction } 
			resource_stockpile_compare = { 
				resource = diversity_of_discussion 
				value >= 60 
			} 
		}
	}
	
	immediate = {
		add_modifier = { modifier = high_diversity_of_discussion days = -1 }
		remove_modifier = low_diversity_of_discussion
		remove_modifier = mid_diversity_of_discussion
	}
}

# Mid Diversity of Discussion
country_event = {
	id = diversity_of_discussion.2
	title = diversity_of_discussion.2.title

	hide_window = yes
	is_triggered_only = no
	
	trigger = {
		is_country_type = default
		
		owner = {
			NOT = {
				has_modifier = mid_diversity_of_discussion
			}
			owner = { has_country_flag = first_faction } 
			resource_stockpile_compare = { 
				resource = diversity_of_discussion 
				value < 60 
			} 
			resource_stockpile_compare = { 
				resource = diversity_of_discussion 
				value > 40 
			} 
		}
	}
	
	immediate = {
		add_modifier = { modifier = mid_diversity_of_discussion days = -1 }
		remove_modifier = high_diversity_of_discussion
		remove_modifier = low_diversity_of_discussion
	}
}
