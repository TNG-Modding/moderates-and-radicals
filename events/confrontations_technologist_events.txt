# technologist
country_event = {
    id = confrontations_technologist.1
    hide_window = yes

    is_triggered_only = no

    mean_time_to_happen = { months = 1 }
    trigger = {
        is_variable_set = current_technologist_support
        is_variable_set = current_ideological_purity

        check_variable = {
            which = current_technologist_support
            value > current_ideological_purity
        }
    }

    immediate = {
        export_resource_income_to_variable = { 
            resource = technologist_support 
            variable = current_technologist_support_income
        }

        # One years income
        multiply_variable = {
            which = current_technologist_support_income
            value = 36
        }

        add_resource = { 
            technologist_support = -1
            mult = current_technologist_support_income 
        }

        export_resource_stockpile_to_variable = { 
            resource = technologist_militancy
            variable = current_technologist_militancy
        }

        if = {
            limit = {
                check_variable = {
                    which = current_technologist_militancy
                    value > 90
                }
            }
            every_pop_faction = {
                limit = {
                    OR = {
                        is_pop_faction_type = gov_technologist
                        is_pop_faction_type = technologist_radicals
                        is_pop_faction_type = technologist
                    }
                }
                pop_faction_event = { id = confrontations_technologist.90 }
            }
        }	
        else_if = {
            limit = {
                check_variable = {
                    which = current_technologist_militancy
                    value > 80
                }
            }
            every_pop_faction = {
                limit = {
                    OR = {
                        is_pop_faction_type = gov_technologist
                        is_pop_faction_type = technologist_radicals
                        is_pop_faction_type = technologist
                    }
                }
                pop_faction_event = { id = confrontations_technologist.80 }
            }
        }
        else_if = {
            limit = {
                check_variable = {
                    which = current_technologist_militancy
                    value > 50
                }
            }
            every_pop_faction = {
                limit = {
                    OR = {
                        is_pop_faction_type = gov_technologist
                        is_pop_faction_type = technologist_radicals
                        is_pop_faction_type = technologist
                    }
                }
                pop_faction_event = { id = confrontations_technologist.30 }
            }
        }
        else_if = {
            limit = {
                check_variable = {
                    which = current_technologist_militancy
                    value > 25
                }
            }
            every_pop_faction = {
                limit = {
                    OR = {
                        is_pop_faction_type = gov_technologist
                        is_pop_faction_type = technologist_radicals
                        is_pop_faction_type = technologist
                    }
                }
                pop_faction_event = { id = confrontations_technologist.20 }
            }
        }
        else = {
            every_pop_faction = {
                limit = {
                    OR = {
                        is_pop_faction_type = gov_technologist
                        is_pop_faction_type = technologist_radicals
                        is_pop_faction_type = technologist
                    }
                }
                pop_faction_event = { id = confrontations_technologist.10 }
            }
        }	 
    }
}

    show_sound = event_factions

    immediate = {
        owner = { country_event = { id = confrontations.1 } }
    }

    option = {
        name = confrontations_technologist.10.a

        owner = {
            export_resource_income_to_variable = {
                resource = alloys
                variable = current_monthly_technologist_alloys_income
            }
            multiply_variable = {
                which = current_monthly_technologist_alloys_income
                value = 6
            }
            
            add_resource = {
                alloys = 1
                mult = current_monthly_technologist_alloys_income
            }

                modifier = technologist_debates_continue_modifier

            }

    }

        name = confrontations_technologist.10.b

        owner = {
            add_modifier = {
                modifier = technologist_debates_stopped_modifier

            }
        }
    }
}

pop_faction_event = {
    id = confrontations_technologist.90

    is_triggered_only = yes

    picture = GFX_evt_open_revolt
    show_sound = event_super_explosion

    immediate = {
        owner = { country_event = { id = confrontations.1 } }
    }

    option = {
        name = confrontations_technologist.90.a
        
        owner = {
            if = {
                limit = { 
                    NOT = { 
                        has_ethic = "ethic_fanatic_materialist" 
                    } 
                }
                shift_ethic = ethic_materialist
            }
            
            add_resource = {
                ideological_purity = -25
            }

            add_modifier = {
                modifier = technologist_coup_continue_modifier

            }

            # Kill ruler
            kill_leader = {
                type = ruler
            }

            # Kill rulers of wrong ethic
            if = {
                limit = {
                    count_owned_pop = {
                        count >= 3
                        limit = {
                            NOT = { pop_has_ethic = ethic_materialist }
                            is_pop_category = ruler
                        }
                    }
                }
                random_owned_pop = {
                    limit = {
                        NOT = { pop_has_ethic = ethic_materialist }
                        is_pop_category = ruler
                    }
                    kill_pop = yes
                }
                random_owned_pop = {
                    limit = {
                        NOT = { pop_has_ethic = ethic_materialist }
                        is_pop_category	= ruler
                    }
                    kill_pop = yes
                }
                random_owned_pop = {
                    limit = {
                        NOT = { pop_has_ethic = ethic_materialist }
                        is_pop_category	= ruler
                    }
                    kill_pop = yes
                }
            }
            else = {

                }
