namespace = technologist_militancy_desired_resource_events

country_event = {
	id = technologist_militancy_desired_resource_events.1
	hide_window = yes
	is_triggered_only = no

	mean_time_to_happen = {
		days = 30
	}

	immediate = {
		# Get the current production
		export_resource_income_to_variable = {
			resource = technologist_militancy_desired_value_change
			variable = current_technologist_militancy_desired_value_change
		}

		# Define new desired point
		set_variable = {
			which = technologist_militancy_new_desired_value
			value = 5
		}
		change_variable = {
			which = technologist_militancy_new_desired_value
			value = current_technologist_militancy_desired_value_change
		}
		# technologist_militancy_new_desired_value is now the new desired point
		
		# Old desired value
		export_resource_stockpile_to_variable = {
			resource = technologist_militancy_desired_value
			variable = current_technologist_militancy_desired_value
		}

		# Determine the delta needed to achieve new desired value
		set_variable = {
			which = technologist_militancy_desired_value_delta
			value = technologist_militancy_new_desired_value 
		}
		subtract_variable = {
			which = technologist_militancy_desired_value_delta
			value = current_technologist_militancy_desired_value
		}
		# technologist_militancy_desired_value_delta is now the delta between new and old desired point

		# Set desired value to new desired value
		add_resource = {
			technologist_militancy_desired_value = 1
			mult = technologist_militancy_desired_value_delta
		}

		# Get current value of technologist_militancy
		export_resource_stockpile_to_variable = {
			resource = technologist_militancy_current_value
			variable = current_technologist_militancy_current_value
		}

		set_variable = {
			which = delta_between_current_desired_technologist_militancy
			value = technologist_militancy_new_desired_value
		}
		subtract_variable = {
			which = technologist_militancy_new_desired_value
			value = current_technologist_militancy_current_value
		}
		# delta_between_current_desired_technologist_militancy now represents the difference between the values

		# If we are close, remove modifiers, if we are far, add trending up or trending down
		if = {
			limit = {
				check_variable = {
					which = delta_between_current_desired_technologist_militancy
					value <= 1
				}
				check_variable = {
					which = delta_between_current_desired_technologist_militancy
					value >= -1
				}
			}
			remove_modifier = technologist_militancy_current_value_trending_up
			remove_modifier = technologist_militancy_current_value_trending_down
		}
		else = {
			if = {
				limit = {
					check_variable = {
						which = technologist_militancy_new_desired_value
						value >= current_technologist_militancy_current_value
					}
				}
				add_modifier = {
					modifier = technologist_militancy_current_value_trending_up
					days = -1
				}
			}
			else = {
				add_modifier = {
					modifier = technologist_militancy_current_value_trending_down
					days = -1
				}
			}
		}
	}	
}
