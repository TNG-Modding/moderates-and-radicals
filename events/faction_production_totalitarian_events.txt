# totalitarian Faction Support Approval Effects
###############################

namespace = totalitarian_support_approval_effects

# Factions despawn, remove modifiers
country_event = {
	id = totalitarian_support_approval_effects.1
	hide_window = yes
	is_triggered_only = no

	mean_time_to_happen = {
		days = 30
	}

	trigger = {
		NOT = {
			any_pop_faction = {
				OR = {
					# is_pop_faction_type = gov_totalitarian
					is_pop_faction_type = totalitarian
					is_pop_faction_type = totalitarian_radicals
				}
			}
		}
	}

	immediate = {
		remove_modifier = totalitarian_effects_powerful_happy
		remove_modifier = totalitarian_effects_weak_happy
		remove_modifier = totalitarian_effects_pathetic_happy
		remove_modifier = totalitarian_effects_powerful_unhappy
		remove_modifier = totalitarian_effects_weak_unhappy
		remove_modifier = totalitarian_effects_pathetic_unhappy
	}	
}

# Check influence of faction
pop_faction_event = {
	id = totalitarian_support_approval_effects.2

	hide_window = yes
	is_triggered_only = no

	mean_time_to_happen = {
		days = 30
	}

	trigger = {
		owner = {
			is_country_type = default
			has_modifier = politics
		}
		OR = {
			# is_pop_faction_type = gov_totalitarian
			is_pop_faction_type = totalitarian
			is_pop_faction_type = totalitarian_radicals
		}
	}

	immediate = {
		owner = {
			if = {
				limit = {
					root = { support >= 0.3 }
				}
				if = {
					limit = {
						root = { faction_approval >= 0.4 }
					}
					if = {
						limit = {
							NOT = { has_modifier = totalitarian_effects_powerful_happy }
						}
						remove_modifier = totalitarian_effects_weak_happy
						remove_modifier = totalitarian_effects_pathetic_happy
						remove_modifier = totalitarian_effects_powerful_unhappy
						remove_modifier = totalitarian_effects_weak_unhappy
						remove_modifier = totalitarian_effects_pathetic_unhappy
						add_modifier = {
							modifier = totalitarian_effects_powerful_happy
							days = -1
						}
					}
				}
				else = {
					if = {
						limit = {
							NOT = { has_modifier = totalitarian_effects_powerful_unhappy }
						}
						remove_modifier = totalitarian_effects_powerful_happy
						remove_modifier = totalitarian_effects_weak_happy
						remove_modifier = totalitarian_effects_pathetic_happy
						remove_modifier = totalitarian_effects_weak_unhappy
						remove_modifier = totalitarian_effects_pathetic_unhappy
						add_modifier = {
							modifier = totalitarian_effects_powerful_unhappy
							days = -1
						}
					}
				}
			}
			else_if = {
				limit = {
					root = { support >= 0.2 }
				}

				if = {
					limit = {
						root = { faction_approval >= 0.4 }
					}
					if = {
						limit = {
							NOT = { has_modifier = totalitarian_effects_weak_happy }
						}
						remove_modifier = totalitarian_effects_powerful_happy
						remove_modifier = totalitarian_effects_pathetic_happy
						remove_modifier = totalitarian_effects_powerful_unhappy
						remove_modifier = totalitarian_effects_weak_unhappy
						remove_modifier = totalitarian_effects_pathetic_unhappy
						add_modifier = {
							modifier = totalitarian_effects_weak_happy
							days = -1
						}
					}
				}
				else = {
					if = {
						limit = {
							NOT = { has_modifier = totalitarian_effects_weak_unhappy }
						}
						remove_modifier = totalitarian_effects_powerful_happy
						remove_modifier = totalitarian_effects_weak_happy
						remove_modifier = totalitarian_effects_pathetic_happy
						remove_modifier = totalitarian_effects_powerful_unhappy
						remove_modifier = totalitarian_effects_pathetic_unhappy
						add_modifier = {
							modifier = totalitarian_effects_weak_unhappy
							days = -1
						}
					}
				}
			}
			else = {
				if = {
					limit = {
						root = { faction_approval > 0.5 }
					}
					if = {
						limit = {
							NOT = { has_modifier = totalitarian_effects_pathetic_happy }
						}
						remove_modifier = totalitarian_effects_powerful_happy
						remove_modifier = totalitarian_effects_weak_happy
						remove_modifier = totalitarian_effects_powerful_unhappy
						remove_modifier = totalitarian_effects_weak_unhappy
						remove_modifier = totalitarian_effects_pathetic_unhappy
						add_modifier = {
							modifier = totalitarian_effects_pathetic_happy
							days = -1
						}
					}
				}
				else = {
					if = {
						limit = {
							NOT = { has_modifier = totalitarian_effects_pathetic_unhappy }
						}
						remove_modifier = totalitarian_effects_powerful_happy
						remove_modifier = totalitarian_effects_weak_happy
						remove_modifier = totalitarian_effects_pathetic_happy
						remove_modifier = totalitarian_effects_powerful_unhappy
						remove_modifier = totalitarian_effects_weak_unhappy
						add_modifier = {
							modifier = totalitarian_effects_pathetic_unhappy
							days = -1
						}
					}
				}
			}
		}
	}
}
