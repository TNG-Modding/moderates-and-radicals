# Traditionalist_radicals Faction

traditionalist_radicals = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_spiritualist

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_radical_factions
		produces = {
			traditionalist_militancy = -12
			traditionalist_support = 12
		}
	}

	is_potential = {
		#years_passed > 11
		#has_encountered_other_species = yes
		owner = { has_country_flag = first_faction } 
		OR = { 
			has_ethic = ethic_fanatic_materialist
			has_ethic = ethic_materialist
		}
		count_owned_pop = {
			count >= 5
			limit = {
				has_ethic = ethic_spiritualist
			}
		}
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	on_create = {
		pop_faction_event = { id = faction_spawn.2 }
		pop_faction_event = { id = faction_spawn.3 }
	}
	on_destroy = {
		# Removing faction exist static modifier
		remove_modifier = traditionalist_radicals_exist
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
		is_robot_pop = no
	}

	can_join_faction = {
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_spiritualist
			has_ethic = ethic_fanatic_spiritualist
		}
	}

	attraction = {
		base = 150

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
	}

	leader = {
		base = 100
		modifier = {
			factor = 0
			species = {
				has_trait = "trait_mechanical"
			}
		}

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}

	demand = { # "TRADITIONALIST_SPIRITUALIST_POPS"
		title = "TRADITIONALIST_SPIRITUALIST_POPS"
		unfulfilled_title = "TRADITIONALIST_FEW_SPIRITUALIST_POPS"
		desc = "TRADITIONALIST_SPIRITUALIST_POPS_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			exists = owner
			owner = {
				pop_percentage = {
					percentage > 0.09
					limit = { pop_has_ethic = ethic_spiritualist }
				}
			}
		}

		trigger = {
			owner = {
				pop_percentage = {
					percentage > 0.24
					limit = { pop_has_ethic = ethic_spiritualist }
				}
			}
		}
	}

	demand = { # "TRADITIONALIST_UNSPIRITUALIST"
		title = "TRADITIONALIST_UNSPIRITUALIST"
		unfulfilled_title = "TRADITIONALIST_NOT_UNSPIRITUALIST"
		desc = "TRADITIONALIST_UNSPIRITUALIST_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				pop_percentage = {
					percentage < 0.25
					limit = { pop_has_ethic = ethic_spiritualist }
				}
			}
		}

		trigger = {
			owner = {
				pop_percentage = {
					percentage > 0.09
					limit = { pop_has_ethic = ethic_spiritualist }
				}
			}
		}
	}
}
