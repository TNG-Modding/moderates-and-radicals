# Techno-Spiritual Moderates
##############################

techno_spiritual_moderate = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_spiritualist

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_moderate_factions
		produces = {
			influence = 1
		}
	}

	is_potential = {
		years_passed > 10
		has_encountered_other_species = yes
		NOT = { has_ethic = ethic_gestalt_consciousness }
		count_owned_pops = {
			count >= 2
			limit = {
				has_ethic = ethic_materialist
			}
		}
		count_owned_pops = {
			count >= 2
			limit = {
				has_ethic = ethic_spiritualist
			}
		}
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_spiritualist
			has_ethic = ethic_fanatic_spiritualist
			has_ethic = ethic_materialist
			has_ethic = ethic_fanatic_materialist
		}
	}

	attraction = {
		base = 150

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}

	on_create = {
		pop_faction_event = { id = factions.1 }
	}

	demand = {
		title = "TECHNOLOGIST_TECH_LEAP"
		unfulfilled_title = "TECHNOLOGIST_TECH_LAG"
		desc = "TECHNOLOGIST_TECH_LEAP_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -20

		potential = {
			exists = owner
			owner = {
				any_relation = {
					has_communications = root.owner
					is_country_type = default
					NOT = { has_government = gov_successor_khanate }
				}
				NOR = {
					has_country_flag = primitives_can_into_space
					is_subject = yes
				}
			}
		}

		trigger = {
			owner = {
				NOT = {
					any_relation = {
						has_communications = root.owner
						is_country_type = default
						NOT = { has_government = gov_successor_khanate }
						relative_power = {
							who = root.owner
							category = technology
							value > equivalent
						}
					}
				}
			}
		}
	}

	demand = {
		title = "TECHNOLOGIST_RESEARCH_TREATIES"
		unfulfilled_title = "TECHNOLOGIST_FEW_RESEARCH_TREATIES"
		desc = "TECHNOLOGIST_RESEARCH_TREATIES_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 2
			}
		}

		trigger = {
			owner = {
				OR = {
					AND = {
						has_federation = yes
						federation = {
							has_federation_perk = research_federation_passive
						}
					}
					count_playable_country = {
						count > 2
						limit = {
							exists = this
							root.owner = { has_research_agreement = prev }
						}
					}
				}
			}
		}
	}

	demand = { # "TRADITIONALIST_NO_ROBOT_RIGHTS"
		title = "TRADITIONALIST_NO_ROBOT_RIGHTS"
		unfulfilled_title = "TRADITIONALIST_ROBOT_RIGHTS"
		desc = "TRADITIONALIST_NO_ROBOT_RIGHTS_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				can_set_ai_policy = yes
				NOT = { has_policy_flag = ai_full_rights }
			}
		}

		trigger = {
			owner = {
				NOT = { has_policy_flag = ai_servitude }
			}
		}
	}

	demand = { # "TRADITIONALIST_NO_FULL_ROBOT_RIGHTS"
		title = "TRADITIONALIST_NO_FULL_ROBOT_RIGHTS"
		unfulfilled_title = "TRADITIONALIST_FULL_ROBOT_RIGHTS"
		desc = "TRADITIONALIST_NO_FULL_ROBOT_RIGHTS_DESC"

		unfulfilled_effect = -20
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				can_set_ai_policy = yes
				has_policy_flag = ai_full_rights
			}
		}

		trigger = {
			owner = {
				NOT = { has_policy_flag = ai_full_rights }
			}
		}
	}

	demand = { # "TRADITIONALIST_NO_TOMB_WORLDS"
		title = "TRADITIONALIST_NO_TOMB_WORLDS"
		unfulfilled_title = "TRADITIONALIST_TOMB_WORLDS"
		desc = "TRADITIONALIST_NO_TOMB_WORLDS_DESC"

		unfulfilled_effect = -5
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				has_surveyed_class = pc_nuked
				NOR = {
					has_origin = origin_post_apocalyptic
					has_country_flag = worm_tomb_ending
					has_valid_civic = civic_memorialist
				}
			}
		}

		trigger = {
			owner = {
				NOR = {
					any_owned_planet = {
						is_planet_class = pc_nuked
					}
					any_planet_within_border = {
						is_planet_class = pc_nuked
						is_terraforming = yes
					}
				}
			}
		}
	}

	demand = { # "TRADITIONALIST_SPIRITUALIST_POPS"
		title = "TRADITIONALIST_SPIRITUALIST_POPS"
		unfulfilled_title = "TRADITIONALIST_FEW_SPIRITUALIST_POPS"
		desc = "TRADITIONALIST_SPIRITUALIST_POPS_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			exists = owner
			owner = {
				pop_percentage = {
					percentage > 0.09
					limit = { pop_has_ethic = ethic_spiritualist }
				}
			}
		}

		trigger = {
			owner = {
				pop_percentage = {
					percentage > 0.24
					limit = { pop_has_ethic = ethic_spiritualist }
				}
			}
		}
	}

	demand = { # "TRADITIONALIST_UNSPIRITUALIST"
		title = "TRADITIONALIST_UNSPIRITUALIST"
		unfulfilled_title = "TRADITIONALIST_NOT_UNSPIRITUALIST"
		desc = "TRADITIONALIST_UNSPIRITUALIST_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				pop_percentage = {
					percentage < 0.25
					limit = { pop_has_ethic = ethic_spiritualist }
				}
			}
		}

		trigger = {
			owner = {
				pop_percentage = {
					percentage > 0.09
					limit = { pop_has_ethic = ethic_spiritualist }
				}
			}
		}
	}
}
