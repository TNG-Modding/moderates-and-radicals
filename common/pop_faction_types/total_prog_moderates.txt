##############################
# Totalitarian Progressive Moderates
##############################

total_prog_moderate = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_egalitarian

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_moderate_factions
		produces = {
			influence = 1
		}
	}

	is_potential = {
		years_passed > 10
		has_encountered_other_species = yes
		NOT = { has_ethic = ethic_gestalt_consciousness }
		count_owned_pops = {
			count >= 2
			limit = {
				has_ethic = ethic_authoritarian
			}
		}
		count_owned_pops = {
			count >= 2
			limit = {
				has_ethic = ethic_egalitarian
			}
		}
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_egalitarian
			has_ethic = ethic_fanatic_egalitarian
			has_ethic = ethic_authoritarian
			has_ethic = ethic_fanatic_authoritarian
		}
	}

	attraction = {
		base = 150

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}

	on_create = {
		pop_faction_event = { id = factions.1 }
	}

	demand = {
		title = "TOTAL_PROG_BALANCED_GOVERNMENT"
		unfulfilled_title = "TOTAL_PROG_UNBALANCED_GOVERNMENT"
		desc = "TOTAL_PROG_BALANCED_GOVERNMENT_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -20

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				OR = {
					has_authority = auth_oligarchic
					has_authority = auth_dictatorial
				}
			}
		}
	}

	demand = {
		title = "PROGRESSIVE_FREE_MOVEMENT"
		unfulfilled_title = "PROGRESSIVE_NO_FREE_MOVEMENT"
		desc = "PROGRESSIVE_FREE_MOVEMENT_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = { has_encountered_other_species = yes }
		}

		trigger = {
			owner = {
				NOR = {
					any_owned_pop = {
						has_migration_control = yes
						is_sapient = yes
						is_enslaved = no
						is_being_purged = no
						is_shackled_robot = no
						is_robot_pop = no
					}
					has_policy_flag = resettlement_allowed
				}
			}
		}
	}

	demand = {
		title = "TOTALITARIAN_EXTRANATIONAL_AUTHORITY"
		unfulfilled_title = "TOTALITARIAN_NO_EXTRANATIONAL_AUTHORITY"
		desc = "TOTALITARIAN_EXTRANATIONAL_AUTHORITY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 2
			}
		}

		trigger = {
			owner = {
				is_overlord = yes
			}
		}
	}

	demand = {
		title = "TOTALITARIAN_UNDERLING"
		unfulfilled_title = "TOTALITARIAN_NOT_UNDERLING"
		desc = "TOTALITARIAN_UNDERLING_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				is_subject = yes
			}
		}

		trigger = {
			owner = {
				is_subject = yes
			}
		}
	}

	demand = {
		title = "PROGRESSIVE_REPRODUCTIVE_FREEDOM"
		unfulfilled_title = "PROGRESSIVE_NO_REPRODUCTIVE_FREEDOM"
		desc = "PROGRESSIVE_REPRODUCTIVE_FREEDOM_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {
			exists = owner
		}

		trigger = {
			NOT = {
				owner = { has_policy_flag = population_controls_allowed }
			}
		}
	}
}
