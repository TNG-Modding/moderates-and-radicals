# Xenoist_radicals Faction
##############################

xenoist_radicals = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_xenophile

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_radical_factions
		produces = {
			influence = -6
			idealogical_purity = -2
		}
	}

	is_potential = {
		years_passed > 11 
		has_encountered_other_species = yes
		owner = { has_country_flag = first_faction } 
		OR = { 
			has_ethic = ethic_fanatic_xenophobe
			has_ethic = ethic_xenophobe
		}
		count_owned_pops = {
			count >= 5
			limit = {
				has_ethic = ethic_xenophile
			}
		}
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	on_create = {
		pop_faction_event = { id = factions.9990 }
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			has_ethic = ethic_xenophile
			has_ethic = ethic_fanatic_xenophile
		}
	}

	attraction = {
		base = 150

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}

		modifier = {
			factor = 2.0
			NOT = { is_same_species = owner_species }
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.5
			leader_class = scientist
		}

		modifier = {
			factor = 2.0
			NOT = { is_same_species = owner_species }
		}

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}
	
	demand = {
		title = "RADICAL_XENOIST_GALVANIZE_RULER_SAME_SPECIES"
		unfulfilled_title = "RADICAL_XENOIST_GALVANIZE_RULER_NOT_SAME_SPECIES"
		desc = "RADICAL_XENOIST_GALVANIZE_RULER_NOT_SAME_SPECIES_DESC"

		fulfilled_effect = 0
		unfulfilled_effect = -50

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				exists = ruler
				ruler = { is_same_species = owner_species }
			}
		}
	}
	
}
