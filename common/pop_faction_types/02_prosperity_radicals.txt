# Prosperity_radicals Faction
##############################

prosperity_radicals = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_pacifist

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_radical_factions
		produces = {
			unity = -6
			idealogical_purity = -2
		}
	}

	is_potential = {
		years_passed > 11
		has_encountered_other_species = yes
		owner = { has_country_flag = first_faction } 
		OR = { 
			has_ethic = ethic_fanatic_militarist
			has_ethic = ethic_militarist
		}
		count_owned_pops = {
			count >= 5
			limit = {
				has_ethic = ethic_pacifist
			}
		}
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	on_create = {
		pop_faction_event = { id = factions.9990 }
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		NOT = { has_trait = trait_nerve_stapled }
		has_ethic = ethic_pacifist
	}

	attraction = {
		base = 100

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}

		modifier = {
			factor = 1.5

			OR = {
				has_ethic = ethic_pacifist
				has_ethic = ethic_fanatic_pacifist
			}
		}

		# Should really check for citizenship instead
		modifier = {
			factor = 1.25
			NOT = { is_same_species = owner_species }
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			OR = {
				leader_class = admiral
				leader_class = general
			}
		}

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}

	demand = {
		title = "PROSPERITY_PEACE"
		unfulfilled_title = "PROSPERITY_NO_PEACE"
		desc = "PROSPERITY_PEACE_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -10

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				is_at_war = no
			}
		}
	}
}
